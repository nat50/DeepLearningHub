📚 Image Size Distribution Analysis

What:

Examines width, height, and aspect ratios of all images in the dataset.

Why:

Understanding size patterns helps optimize preprocessing, identify outliers, and plan model input requirements.

How:

Calculate statistics for width/height/aspect ratios, visualize distributions with histograms and a scatter plot.

from pathlib import Path

import pandas as pd
from PIL import Image
import plotly.express as px

IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".bmp", ".webp"}
IMG_DIR = Path("data/food-101/images")


def iter_image_paths(img_dir: Path):
    for class_dir in sorted([path for path in img_dir.iterdir() if path.is_dir()]):
        for path in sorted(class_dir.iterdir()):
            if path.is_file() and path.suffix.lower() in IMAGE_EXTENSIONS:
                yield class_dir.name, path


rows = []
for class_name, path in iter_image_paths(IMG_DIR):
    with Image.open(path) as img:
        rgb = img.convert("RGB")
        width, height = rgb.size
    rows.append(
        {
            "class_name": class_name,
            "path": str(path),
            "width": width,
            "height": height,
            "file_size_kb": path.stat().st_size / 1024.0,
            "aspect_ratio": width / max(height, 1),
        }
    )

records = pd.DataFrame(rows)

px.scatter(
    records.sample(min(3000, len(records)), random_state=42),
    x="width",
    y="height",
    opacity=0.6,
    title="Image Size Distribution (Width vs Height)",
).show()

px.histogram(
    records,
    x="file_size_kb",
    nbins=40,
    title="File Size Distribution (KB)",
    labels={"file_size_kb": "File size (KB)"},
).show()

px.histogram(
    records,
    x="aspect_ratio",
    nbins=40,
    title="Aspect Ratio Distribution",
    labels={"aspect_ratio": "Width / Height"},
).show()

📚 Color Quality Analysis

What:

Analysis of RGB distribution and sharpness/blur metrics.

Why:

Color biases can affect CNNs, while blur metrics guide data cleaning.

How:

Plot 3D RGB scatter, calculate Laplacian variance for sharpness.

from pathlib import Path

import cv2
import numpy as np
import pandas as pd
from PIL import Image
import plotly.express as px
import plotly.graph_objects as go
from plotly.subplots import make_subplots

IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".bmp", ".webp"}
IMG_DIR = Path("data/food-101/images")


def read_image_rgb_uint8(path: str, image_size: int | None = None) -> np.ndarray:
    with Image.open(path) as img:
        rgb = img.convert("RGB")
        if image_size is not None:
            rgb = rgb.resize((image_size, image_size))
        return np.asarray(rgb, dtype=np.uint8)


def iter_image_paths(img_dir: Path):
    for class_dir in sorted([path for path in img_dir.iterdir() if path.is_dir()]):
        for path in sorted(class_dir.iterdir()):
            if path.is_file() and path.suffix.lower() in IMAGE_EXTENSIONS:
                yield class_dir.name, path


rows = []
for class_name, path in iter_image_paths(IMG_DIR):
    with Image.open(path) as img:
        rgb = img.convert("RGB")
        arr = np.asarray(rgb, dtype=np.float32) / 255.0
    gray = cv2.cvtColor((arr * 255).astype(np.uint8), cv2.COLOR_RGB2GRAY)
    rows.append(
        {
            "class_name": class_name,
            "path": str(path),
            "brightness": float(arr.mean()),
            "contrast": float(arr.std()),
            "sharpness": float(cv2.Laplacian(gray, cv2.CV_64F).var()),
        }
    )

records = pd.DataFrame(rows)

rng = np.random.default_rng(42)
sampled_records = records.sample(min(120, len(records)), random_state=42)
rgb_frames = []
for _, row in sampled_records.iterrows():
    pixels = read_image_rgb_uint8(row["path"], image_size=128).reshape(-1, 3)
    if len(pixels) > 256:
        pixels = pixels[rng.choice(len(pixels), size=256, replace=False)]
    pixel_frame = pd.DataFrame(pixels, columns=["red", "green", "blue"])
    pixel_frame["brightness"] = pixel_frame[["red", "green", "blue"]].mean(axis=1).round(1)
    pixel_frame["class_name"] = row["class_name"]
    pixel_frame["rgb_color"] = pixel_frame.apply(
        lambda px_row: f"rgb({px_row['red']},{px_row['green']},{px_row['blue']})",
        axis=1,
    )
    rgb_frames.append(pixel_frame)

rgb_plot = pd.concat(rgb_frames, ignore_index=True)
go.Figure(
    data=[
        go.Scatter3d(
            x=rgb_plot["red"],
            y=rgb_plot["green"],
            z=rgb_plot["blue"],
            mode="markers",
            marker=dict(size=3, opacity=0.5, color=rgb_plot["rgb_color"]),
            customdata=np.stack([rgb_plot["class_name"], rgb_plot["brightness"]], axis=1),
            hovertemplate="Class: %{customdata[0]}
R: %{x}
G: %{y}
B: %{z}
Brightness: %{customdata[1]}",
        )
    ]
).show()

px.scatter(
    records.sample(min(4000, len(records)), random_state=42),
    x="sharpness",
    y="contrast",
    color="class_name",
    opacity=0.65,
    title="Image Quality Metrics",
    hover_data={"brightness": ":.1f", "sharpness": ":.1f", "contrast": ":.1f", "path": False},
).show()

px.histogram(records, x="sharpness", nbins=40, title="Sharpness Distribution").show()

blurry = records.sort_values("sharpness").head(5)
sharp = records.sort_values("sharpness").tail(5)
subplot_titles = []
for label, frame in [("Blurriest", blurry), ("Sharpest", sharp)]:
    for _, row in frame.iterrows():
        subplot_titles.append(f"{label}
{row['class_name']}
sharp={row['sharpness']:.1f}")

fig = make_subplots(rows=2, cols=5, subplot_titles=subplot_titles, vertical_spacing=0.08)
for row_idx, frame in enumerate([blurry, sharp], start=1):
    for col_idx, (_, row) in enumerate(frame.iterrows(), start=1):
        fig.add_trace(go.Image(z=read_image_rgb_uint8(row["path"], image_size=256)), row=row_idx, col=col_idx)
fig.update_layout(title="Blurriest vs Sharpest Samples", height=700, width=1300)
fig.show()
