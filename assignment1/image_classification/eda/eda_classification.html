101-class fine-grained food classification task — class distribution, splits, hierarchy, and feature analysis

📊 Classification Overview

▼

Classification overview

What:

Total images, class count, balance ratio, and min/max images per class.

Why:

Confirms scale and whether classes are balanced before modeling.

How:

Aggregate records with build_classification_summary (see notebook).

from pathlib import Path

import cv2
import numpy as np
import pandas as pd
from PIL import Image

IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".bmp", ".webp"}
IMG_DIR = Path("data/food-101/images")


def iter_image_paths(img_dir: Path):
    for class_dir in sorted([path for path in img_dir.iterdir() if path.is_dir()]):
        for path in sorted(class_dir.iterdir()):
            if path.is_file() and path.suffix.lower() in IMAGE_EXTENSIONS:
                yield class_dir.name, path


rows = []
for class_name, path in iter_image_paths(IMG_DIR):
    with Image.open(path) as img:
        rgb = img.convert("RGB")
        arr = np.asarray(rgb, dtype=np.float32) / 255.0
    gray = cv2.cvtColor((arr * 255).astype(np.uint8), cv2.COLOR_RGB2GRAY)
    rows.append(
        {
            "class_name": class_name,
            "path": str(path),
            "file_size_kb": path.stat().st_size / 1024.0,
            "brightness": float(arr.mean()),
            "contrast": float(arr.std()),
            "sharpness": float(cv2.Laplacian(gray, cv2.CV_64F).var()),
        }
    )

records = pd.DataFrame(rows)
class_counts = records.groupby("class_name").size().sort_values(ascending=False)
summary = {
    "total_images": int(len(records)),
    "num_classes": int(records["class_name"].nunique()),
    "max_count": int(class_counts.max()),
    "min_count": int(class_counts.min()),
    "imbalance_ratio": float(class_counts.max() / max(class_counts.min(), 1)),
}

print("CLASSIFICATION EDA REPORT")
print(f"Total images for classification: {summary['total_images']}")
print(f"Classes: {summary['num_classes']}")
print(f"Max count: {summary['max_count']}")
print(f"Min count: {summary['min_count']}")
print(f"Imbalance ratio: {summary['imbalance_ratio']:.2f}x")

📊 Train/Val/Test Split Distribution

▼

Train / validation / test splits

What:

Why:

How:

make_stratified_splits then plot_split_distribution.

from pathlib import Path

import pandas as pd
import plotly.express as px
from sklearn.model_selection import train_test_split

IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".bmp", ".webp"}
IMG_DIR = Path("data/food-101/images")


def iter_image_paths(img_dir: Path):
    for class_dir in sorted([path for path in img_dir.iterdir() if path.is_dir()]):
        for path in sorted(class_dir.iterdir()):
            if path.is_file() and path.suffix.lower() in IMAGE_EXTENSIONS:
                yield class_dir.name, path


records = pd.DataFrame(
    [{"class_name": class_name, "path": str(path)} for class_name, path in iter_image_paths(IMG_DIR)]
)

train_df, temp_df = train_test_split(
    records,
    train_size=0.7,
    random_state=42,
    stratify=records["class_name"],
)
val_df, test_df = train_test_split(
    temp_df,
    train_size=0.5,
    random_state=42,
    stratify=temp_df["class_name"],
)

split_df = records.copy()
split_df["split"] = "unassigned"
split_df.loc[train_df.index, "split"] = "train"
split_df.loc[val_df.index, "split"] = "val"
split_df.loc[test_df.index, "split"] = "test"

overall = split_df["split"].value_counts().reindex(["train", "val", "test"])
px.bar(
    x=overall.index,
    y=overall.values,
    title="Overall Split Distribution",
    labels={"x": "Split", "y": "Image count"},
).show()

top_classes = split_df["class_name"].value_counts().head(10).index
class_split_counts = (
    split_df.loc[split_df["class_name"].isin(top_classes)]
    .groupby(["class_name", "split"])
    .size()
    .reset_index(name="count")
)
px.bar(
    class_split_counts,
    x="class_name",
    y="count",
    color="split",
    barmode="group",
    title="Top Class Split Distribution",
    labels={"class_name": "Class", "count": "Image count"},
).update_layout(xaxis_tickangle=-45).show()

🔬 Feature Visualization

▼

Feature visualization

What:

ResNet50 (ImageNet) embeddings, PCA variance curve, PCA / t-SNE / UMAP 2D plots.

Why:

See whether classes separate in feature space before training a head.

How:

extract_image_features computes embeddings, then plot_feature_embeddings renders PCA, t-SNE, and UMAP views.

from pathlib import Path

import numpy as np
import pandas as pd
from PIL import Image
import plotly.express as px
from sklearn.decomposition import PCA
from sklearn.manifold import TSNE
import torch
from torchvision import models, transforms

try:
    import umap
except ImportError:
    umap = None

IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".bmp", ".webp"}
IMG_DIR = Path("data/food-101/images")


def iter_image_paths(img_dir: Path):
    for class_dir in sorted([path for path in img_dir.iterdir() if path.is_dir()]):
        for path in sorted(class_dir.iterdir()):
            if path.is_file() and path.suffix.lower() in IMAGE_EXTENSIONS:
                yield class_dir.name, path


records = pd.DataFrame(
    [{"class_name": class_name, "path": str(path)} for class_name, path in iter_image_paths(IMG_DIR)]
)

sampled_frames = []
for _, group in records.groupby("class_name"):
    sampled_frames.append(group.sample(n=min(120, len(group)), random_state=42))
feature_records = (
    pd.concat(sampled_frames, ignore_index=False)
    .sample(frac=1.0, random_state=42)
    .head(1200)
    .reset_index(drop=True)
)

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
weights = models.ResNet50_Weights.DEFAULT
model = models.resnet50(weights=weights)
model.fc = torch.nn.Identity()
model.eval().to(device)

transform = transforms.Compose(
    [
        transforms.Resize((224, 224)),
        transforms.ToTensor(),
        transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225]),
    ]
)

features = []
with torch.no_grad():
    for start_idx in range(0, len(feature_records), 16):
        batch_paths = feature_records["path"].iloc[start_idx : start_idx + 16].tolist()
        batch_tensors = []
        for path in batch_paths:
            with Image.open(path) as img:
                batch_tensors.append(transform(img.convert("RGB")))
        batch = torch.stack(batch_tensors).to(device)
        features.append(model(batch).detach().cpu().numpy())
features = np.vstack(features)

pca_2d = PCA(n_components=2, random_state=42)
pca_points = pca_2d.fit_transform(features)
tsne = TSNE(n_components=2, random_state=42, init="pca", perplexity=min(30, max(2, len(features) // 4), len(features) - 1))
tsne_points = tsne.fit_transform(features)

emb_df = pd.DataFrame(
    {
        "pca_x": pca_points[:, 0],
        "pca_y": pca_points[:, 1],
        "tsne_x": tsne_points[:, 0],
        "tsne_y": tsne_points[:, 1],
        "label": feature_records["class_name"].tolist(),
    }
)
px.scatter(emb_df, x="pca_x", y="pca_y", color="label", title="PCA 2D Projection").show()
px.scatter(emb_df, x="tsne_x", y="tsne_y", color="label", title="t-SNE 2D Projection").show()

if umap is not None:
    umap_points = umap.UMAP(random_state=42).fit_transform(features)
    umap_df = pd.DataFrame({"umap_x": umap_points[:, 0], "umap_y": umap_points[:, 1], "label": feature_records["class_name"].tolist()})
    px.scatter(umap_df, x="umap_x", y="umap_y", color="label", title="UMAP 2D Projection").show()

pca_full = PCA().fit(features)
ev_df = pd.DataFrame(
    {
        "components": np.arange(1, len(pca_full.explained_variance_ratio_) + 1),
        "cum_explained_variance": np.cumsum(pca_full.explained_variance_ratio_),
    }
)
px.line(
    ev_df,
    x="components",
    y="cum_explained_variance",
    markers=True,
    title="PCA Explained Variance",
    labels={"components": "Number of components", "cum_explained_variance": "Cumulative explained variance"},
).show()

🔗 Class Similarity & Clustering

▼

Similarity and clustering

What:

Cosine similarity between class centroids, top pairs, and agglomerative clusters.

Why:

Find easily confused classes and structure for multi-task or hierarchical ideas.

How:

build_similarity_outputs + plot_similarity_outputs (px.imshow heatmap).

from pathlib import Path

import numpy as np
import pandas as pd
from PIL import Image
import plotly.express as px
from sklearn.cluster import AgglomerativeClustering
from sklearn.metrics.pairwise import cosine_similarity
import torch
from torchvision import models, transforms

IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".bmp", ".webp"}
IMG_DIR = Path("data/food-101/images")


def iter_image_paths(img_dir: Path):
    for class_dir in sorted([path for path in img_dir.iterdir() if path.is_dir()]):
        for path in sorted(class_dir.iterdir()):
            if path.is_file() and path.suffix.lower() in IMAGE_EXTENSIONS:
                yield class_dir.name, path


records = pd.DataFrame(
    [{"class_name": class_name, "path": str(path)} for class_name, path in iter_image_paths(IMG_DIR)]
)

sampled_frames = []
for _, group in records.groupby("class_name"):
    sampled_frames.append(group.sample(n=min(120, len(group)), random_state=42))
feature_records = (
    pd.concat(sampled_frames, ignore_index=False)
    .sample(frac=1.0, random_state=42)
    .head(1200)
    .reset_index(drop=True)
)

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
weights = models.ResNet50_Weights.DEFAULT
model = models.resnet50(weights=weights)
model.fc = torch.nn.Identity()
model.eval().to(device)

transform = transforms.Compose(
    [
        transforms.Resize((224, 224)),
        transforms.ToTensor(),
        transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225]),
    ]
)

features = []
with torch.no_grad():
    for start_idx in range(0, len(feature_records), 16):
        batch_paths = feature_records["path"].iloc[start_idx : start_idx + 16].tolist()
        batch_tensors = []
        for path in batch_paths:
            with Image.open(path) as img:
                batch_tensors.append(transform(img.convert("RGB")))
        batch = torch.stack(batch_tensors).to(device)
        features.append(model(batch).detach().cpu().numpy())
features = np.vstack(features)

class_centroids = (
    pd.DataFrame(features)
    .assign(class_name=feature_records["class_name"].values)
    .groupby("class_name")
    .mean()
)
sim = cosine_similarity(class_centroids.values)
sim_df = pd.DataFrame(sim, index=class_centroids.index, columns=class_centroids.index)

pairs = []
for i, row_name in enumerate(sim_df.index):
    for j, col_name in enumerate(sim_df.columns):
        if j <= i:
            continue
        pairs.append((row_name, col_name, float(sim_df.iloc[i, j])))
top_pairs = (
    pd.DataFrame(pairs, columns=["class_1", "class_2", "similarity"])
    .sort_values("similarity", ascending=False)
    .reset_index(drop=True)
)

cluster_labels = AgglomerativeClustering(n_clusters=min(5, len(class_centroids))).fit_predict(class_centroids.values)
cluster_df = pd.DataFrame({"class_name": class_centroids.index, "cluster": cluster_labels + 1})

px.imshow(
    sim_df.values,
    x=sim_df.columns.tolist(),
    y=sim_df.index.tolist(),
    color_continuous_scale="Viridis",
    title="Class Similarity Matrix",
    aspect="auto",
).show()

print(top_pairs.head(10))
print(cluster_df.groupby("cluster")["class_name"].apply(list).reset_index(name="classes"))

Dataset Characteristics

  • Balanced: Food-101 has exactly 1,000 images per class — perfectly balanced
  • Fine-grained: 101 classes of food dishes with visually similar classes
  • Challenging: High intra-class variance and inter-class similarity

Recommendations

  • Transfer Learning: Use pre-trained CNNs (ResNet50, MobileNetV3) or ViTs (ViT, Swin)
  • Data Augmentation: RandomCrop, HFlip, ColorJitter for robustness
  • Expected Accuracy: ResNet50: ~80%, ViT/Swin: ~85-88%
  • Fine-tuning: Frozen backbone + trainable head is effective for this dataset size
